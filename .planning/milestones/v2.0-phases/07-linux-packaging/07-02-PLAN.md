Add a `build-linux` job to `.github/workflows/package.yml` after the existing `build` job. Per D-01, this is a parallel sibling job (NOT matrix, NOT separate workflow). Per D-02, it runs `pnpm run package:nosign`. Per D-03, it includes the same Linux build deps as main.yml. Per D-04, it publishes to both Vortex and Vortex-Staging.

Add the following job after the `build` job (after line 216):

```yaml
  build-linux:
    runs-on: ubuntu-latest
    env:
      NO_PARALLEL: true
      NEXUS_API_KEY: ${{ secrets.NEXUS_API_KEY }}
      DEBUG: electron-builder

    steps:
      - name: Show Inputs
        run: echo "${{ toJSON(github.event.inputs) }}"

      - name: Set Outputs
        id: setOutputs
        env:
          InputVersion: ${{ inputs.version }}
        run: |
          SEMVER_REGEX='^(0|[1-9][0-9]*)\.(0|[1-9][0-9]*)\.(0|[1-9][0-9]*)(-[0-9A-Za-z.-]+)?(\+[0-9A-Za-z.-]+)?$'
          if [[ "$InputVersion" == v* ]]; then
            TAG_VERSION="$InputVersion"
            RAW_VERSION="${InputVersion#v}"
          else
            TAG_VERSION="v$InputVersion"
            RAW_VERSION="$InputVersion"
          fi
          if ! [[ "$RAW_VERSION" =~ $SEMVER_REGEX ]]; then
            echo "Invalid version format. Must be semver."
            exit 1
          fi
          echo "tagVersion=$TAG_VERSION" >> "$GITHUB_OUTPUT"
          echo "rawVersion=$RAW_VERSION" >> "$GITHUB_OUTPUT"
          echo "VORTEX_VERSION=$RAW_VERSION" >> "$GITHUB_ENV"

      - uses: actions/checkout@v6
        with:
          submodules: "recursive"

      - name: Setup Python 3.11
        uses: actions/setup-python@v6
        with:
          python-version: "3.11"

      - name: Install Rust toolchain
        uses: dtolnay/rust-toolchain@stable
        with:
          toolchain: stable

      - name: Install Linux build dependencies
        run: sudo apt-get update && sudo apt-get install -y libfontconfig1-dev cmake liblz4-dev

      - name: Read Node.js version from package.json
        id: node-version
        run: |
          NODE_VERSION=$(node -e "console.log(require('./package.json').engines.node)")
          echo "NODE_VERSION=$NODE_VERSION" >> "$GITHUB_OUTPUT"
          echo "Using Node.js version: $NODE_VERSION"

      - name: Install pnpm
        uses: pnpm/action-setup@v5

      - name: Use Node.js
        uses: actions/setup-node@v6
        with:
          node-version: ${{ steps.node-version.outputs.NODE_VERSION }}
          cache: "pnpm"

      - name: Install dependencies
        run: pnpm install

      - name: Create Unsigned Linux Package
        run: pnpm run package:nosign

      - name: Validate Linux Package Creation
        run: |
          APPIMAGE_EXISTS=$(ls ./dist/vortex-setup-*.AppImage 2>/dev/null | head -1)
          DEB_EXISTS=$(ls ./dist/vortex_*_amd64.deb 2>/dev/null | head -1)
          LATEST_YML_EXISTS=$(ls ./dist/latest-linux.yml 2>/dev/null | head -1)

          if [ -z "$APPIMAGE_EXISTS" ]; then
            echo "ERROR: AppImage not found in ./dist/"
            exit 1
          fi

          if [ -z "$DEB_EXISTS" ]; then
            echo "ERROR: .deb package not found in ./dist/"
            exit 1
          fi

          if [ -z "$LATEST_YML_EXISTS" ]; then
            echo "ERROR: latest-linux.yml not found in ./dist/"
            exit 1
          fi

          echo "Package validation successful - AppImage, .deb, and latest-linux.yml created"

      - name: Create GitHub Release
        uses: softprops/action-gh-release@v2.6.1
        if: ${{ inputs.release == true }}
        with:
          files: |
            ./dist/vortex-setup-*.AppImage
            ./dist/vortex_*_amd64.deb
            ./dist/latest-linux.yml
          prerelease: true
          draft: true
          name: ${{ steps.setOutputs.outputs.rawVersion }}
          tag_name: ${{ steps.setOutputs.outputs.tagVersion }}

      - name: Create GitHub Release on Vortex-Staging
        uses: softprops/action-gh-release@v2.6.1
        if: ${{ inputs.staging-release == true }}
        with:
          files: |
            ./dist/vortex-setup-*.AppImage
            ./dist/vortex_*_amd64.deb
            ./dist/latest-linux.yml
          prerelease: true
          draft: true
          name: ${{ steps.setOutputs.outputs.rawVersion }}
          tag_name: ${{ steps.setOutputs.outputs.tagVersion }}
          token: ${{ secrets.PERSONAL_ACCESS_TOKEN }}
          repository: "Nexus-Mods/Vortex-Staging"

      - name: Create Linux Artifact
        uses: actions/upload-artifact@v7
        if: ${{ inputs.create-artifacts == true }}
        with:
          name: vortex-setup-${{ steps.setOutputs.outputs.rawVersion }}-linux
          path: |
            ./dist/vortex-setup-*.AppImage
            ./dist/vortex_*_amd64.deb
            ./dist/latest-linux.yml
          if-no-files-found: error
```

- Validation step checks for AppImage, .deb, and latest-linux.yml existence before upload
- No `dotnet`, `CodeSignTool`, `VC_redist`, or `windowsdesktop-runtime` steps (Windows-only)
- No `node-gyp VS2022` configuration (Windows-only)
